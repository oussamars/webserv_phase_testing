# 1. Main page (9000) -> 200 OK
curl -i http://localhost:9000/

# 2. Directory listing -> 200 OK
curl -i http://localhost:9000/folder/

# 3. 404 Error -> 404 (custom page)
curl -i http://localhost:9000/does_not_exist

# 4. Method Not Allowed -> 405 (PUT is not allowed on 9001)
curl -i -X PUT http://localhost:9001/index.html

# 5. Create a file -> 201 Created
curl -i -X POST http://localhost:9000/test_life.txt -d "Hello Eval"

# 6. Verify it exists -> 200 OK
curl -i http://localhost:9000/test_life.txt

# 7. Delete it -> 204 No Content
curl -i -X DELETE http://localhost:9000/test_life.txt

# 8. Verify it's gone -> 404 Not Found
curl -i http://localhost:9000/test_life.txt

# 9. CGI execution -> 200 OK (After restoring index.html)
curl -i "http://localhost:9001/index.html?name=evaluation"
